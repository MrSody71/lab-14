"""
Weather Pipeline Dashboard — главная страница Streamlit.

Запуск:
  cd dashboard
  uv run streamlit run src/dashboard/app.py
"""
from __future__ import annotations

import time
from datetime import datetime, timedelta

import polars as pl
import streamlit as st

from dashboard.data import (
    DataSource, CITIES, generate_live_row,
)
from dashboard.components import (
    render_kpi_cards,
    render_temperature_timeline,
    render_current_temps,
    render_humidity_wind_scatter,
    render_comfort_gauges,
    render_data_table,
)

# ── Конфигурация страницы ─────────────────────────────────────────────────

st.set_page_config(
    page_title="Weather Pipeline",
    page_icon="🌤️",
    layout="wide",
    initial_sidebar_state="expanded",
)

# ── Стили ─────────────────────────────────────────────────────────────────

st.markdown("""
<style>
    .main-header {
        font-size: 2rem;
        font-weight: 700;
        color: #1565C0;
        margin-bottom: 0;
    }
    .sub-header {
        color: #666;
        font-size: 0.9rem;
        margin-top: 0;
    }
    .source-badge {
        background: #E3F2FD;
        color: #1565C0;
        padding: 2px 8px;
        border-radius: 12px;
        font-size: 0.8rem;
    }
    [data-testid="stMetricValue"] {
        font-size: 1.4rem;
    }
</style>
""", unsafe_allow_html=True)


# ── State инициализация ───────────────────────────────────────────────────

if "data_source" not in st.session_state:
    st.session_state.data_source = DataSource()

if "df" not in st.session_state:
    st.session_state.df = st.session_state.data_source.load()

if "last_refresh" not in st.session_state:
    st.session_state.last_refresh = datetime.utcnow()

if "live_rows" not in st.session_state:
    st.session_state.live_rows: list[dict] = []


# ── Sidebar ───────────────────────────────────────────────────────────────

with st.sidebar:
    st.title("⚙️ Controls")
    st.markdown("---")

    # Выбор городов
    st.subheader("🌍 Cities")
    available_cities = (
        st.session_state.df["city"].unique().sort().to_list()
        if len(st.session_state.df) > 0
        else CITIES[:5]
    )
    selected_cities = st.multiselect(
        "Select cities to display",
        options=available_cities,
        default=available_cities[:5],
    )

    st.markdown("---")

    # Временной диапазон
    st.subheader("⏱️ Time Range")
    time_range = st.select_slider(
        "Show last N hours",
        options=[1, 3, 6, 12, 24, 48],
        value=1,
    )

    st.markdown("---")

    # Авто-обновление
    st.subheader("🔄 Auto Refresh")
    auto_refresh = st.toggle(
        "Enable live updates", value=True
    )
    refresh_interval = st.slider(
        "Refresh interval (seconds)",
        min_value=2,
        max_value=30,
        value=5,
        disabled=not auto_refresh,
    )

    st.markdown("---")

    # Источник данных
    st.subheader("📡 Data Source")
    src_mode = st.session_state.data_source.mode
    st.markdown(
        f'<span class="source-badge">📦 {src_mode}</span>',
        unsafe_allow_html=True,
    )
    if st.button("🔃 Reload Data", use_container_width=True):
        st.session_state.df = (
            st.session_state.data_source.load()
        )
        st.session_state.last_refresh = datetime.utcnow()
        st.rerun()

    st.markdown("---")

    # Статистика
    df_info = st.session_state.df
    st.caption(
        f"**Records:** {len(df_info):,}  \n"
        f"**Cities:** "
        f"{df_info['city'].n_unique() if len(df_info) > 0 else 0}  \n"
        f"**Last refresh:** "
        f"{st.session_state.last_refresh.strftime('%H:%M:%S')} UTC"
    )


# ── Заголовок ─────────────────────────────────────────────────────────────

col_title, col_status = st.columns([3, 1])
with col_title:
    st.markdown(
        '<p class="main-header">🌤️ Weather Pipeline Dashboard</p>',
        unsafe_allow_html=True,
    )
    st.markdown(
        '<p class="sub-header">'
        'Real-time weather aggregates · '
        'Go collector → Kafka/NATS → Polars → Streamlit'
        '</p>',
        unsafe_allow_html=True,
    )
with col_status:
    status_color = "🟢" if len(st.session_state.df) > 0 else "🔴"
    st.metric(
        "Pipeline",
        f"{status_color} {'Live' if auto_refresh else 'Paused'}",
    )

st.markdown("---")


# ── Основной контент ──────────────────────────────────────────────────────

df = st.session_state.df

# KPI карточки
render_kpi_cards(df)

st.markdown("---")

# Два столбца: timeline + current temps
col_left, col_right = st.columns([2, 1])

with col_left:
    render_temperature_timeline(
        df, selected_cities, hours=time_range
    )

with col_right:
    render_current_temps(df)

st.markdown("---")

# Humidity scatter + Comfort gauges
col_scatter, col_gauges = st.columns([1, 1])

with col_scatter:
    render_humidity_wind_scatter(df)

with col_gauges:
    render_comfort_gauges(df, selected_cities[:4])

st.markdown("---")

# Таблица данных (сворачиваемая)
with st.expander("📋 Raw Data Table", expanded=False):
    render_data_table(df, n_rows=50)

# Footer
st.markdown("---")
st.caption(
    "Weather Pipeline · Lab Work #14 · "
    "Go + Rust + Python · "
    f"Data: {len(df):,} records"
)


# ── Live обновление ───────────────────────────────────────────────────────

if auto_refresh:
    # Добавляем новые mock-строки к существующим данным
    new_rows = st.session_state.data_source.load_live_update()

    if len(new_rows) > 0:
        combined = pl.concat(
            [st.session_state.df, new_rows],
            how="diagonal",
        )
        # Ограничиваем размер: последние 5000 строк
        if len(combined) > 5000:
            combined = combined.tail(5000)
        st.session_state.df = combined
        st.session_state.last_refresh = datetime.utcnow()

    time.sleep(refresh_interval)
    st.rerun()
